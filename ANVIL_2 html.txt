            function applyFilters() {

                        clampVisible = !ss_Clamp_SH_Numbers.includes(shValue);

                    } else {

                });

            function calculateAngularity() {
                const l = parseFloat(lengthInput.value) || 0;
                const dx = parseFloat(displacementXInput.value) || 0;
                const dy = parseFloat(displacementYInput.value) || 0;
                const dz = parseFloat(displacementZInput.value) || 0;

                const totalDisplacement = Math.sqrt(Math.pow(dx, 2) + Math.pow(dy, 2) + Math.pow(dz, 2));

                if (l > 0 && totalDisplacement > 0) {

                }

        function getLoadColumn(tempF) {
            return tempF <= 650 ? 'maxLoad650' : 'maxLoad750';

        function findComponentByLoad(dataArray, load, loadColumnKey) {
            if (!dataArray || !Array.isArray(dataArray)) return null;
            return dataArray.find(item => item[loadColumnKey] !== null && item[loadColumnKey] !== undefined && item[loadColumnKey] >= load);
        }
        
        function findGenericComponentByLoad(dataArray, load, key = 'maxLoad') {
             if (!dataArray || !Array.isArray(dataArray)) return null;
             return dataArray.find(item => item[key] !== null && item[key] !== undefined && item[key] >= load);
        }

        function getPipeComponent(dataObject, pipeSize) {
            if (dataObject && dataObject[pipeSize]) {
                const componentData = dataObject[pipeSize];
                // Create a copy and add the 'pipeSize' property for display purposes
                return { ...componentData, pipeSize: pipeSize };
            }
            return null;
        }

        function generateValidationMessage(componentData, load, loadColumnKey, pipeSize, genericLoadKey = 'maxLoad') {
            if (!componentData) {
                return `<p class="invalid-msg">No data for pipe size ${pipeSize}".</p>`;
            }
            const maxLoad = componentData[loadColumnKey] !== undefined ? componentData[loadColumnKey] : componentData[genericLoadKey];
            
            if (maxLoad === null || maxLoad === undefined) {
                return `<p class="invalid-msg">No load data available for this temperature.</p>`;
            }
            if (load > maxLoad) {
                return `<p class="invalid-msg">Load of ${load} lbs exceeds max of ${maxLoad} lbs.</p>`;
            }
            return `<p class="valid-msg">Suitable for the load.</p>`;
        }
        
        function generateSpecTable(data, specMap, validationMessage = '') {
            let content = validationMessage;
            if (!data) {
                content += 'Component not found for the given requirements.';
            } else {
                let tableHtml = '<table style="width:100%; border:none;">';
                for (const [label, key] of Object.entries(specMap)) {
                    const value = key.split('.').reduce((o, i) => (o ? o[i] : undefined), data); // Handle nested keys like data.key
                    tableHtml += `<tr><td style="text-align:left; font-weight:bold;">${label}:</td><td style="text-align:left;">${value !== undefined && value !== null ? value : 'N/A'}</td></tr>`;
                }
                tableHtml += '</table>';
                content += tableHtml;
            }
            return `<td>${content}</td>`;
        }

        function generateComponentColumn(title, imageUrl, data, specMap, validationMessage = '') {
            let content = `<img src="${imageUrl}" alt="${title}"><div class="section-title">${title}</div>`;
            return `<td>${content}${generateSpecTable(data, specMap, validationMessage)}</td>`;
        }
